linux任务按秒执行

January 3, 2024 · 1 min · ZhaoGuibin 
#!/bin/bash
path=$(dirname "$PWD")
php_bin="/usr/local/php/bin/php"

while true; do
    cd $path && $php_bin think crontab_1 && $php_bin think crontab_2
    # 等待五秒钟
    sleep 5
done
