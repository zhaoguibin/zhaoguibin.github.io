多维数组拼接成字符串#
  
    /**
     * @FuncName:arrayToString
     * @Description 数组拼接成字符串
     * @User ZHaoGuiBin
     * @Date 2020-08-14 9:17
     * @param $data
     * @param string $keys
     * @param string $key_1 //多维数组主key
     * @param string $string
     * @return string
     */
    private static function arrayToString($data, $keys = '', $string = '', $key_1 = '')
    {
        $temp = 0;
        foreach ($data as $key => $value) {
            if (is_array($value)) {
                if (!is_numeric($key)) {
                    $key_1 = $key;
                    $string .= $key_1 . '=';
                }
                $string .= self::arrayToString($value, $key, '', $key_1);
            } else {
                if ($keys) {
                    if (is_numeric($key)) {
                        if ($temp == 0) {
                            $string .= $keys . '=' . $value;
                            $temp++;
                        } else {
                            $string .= '&' . $value . '&';
                        }
                    } else {
                        if ($temp == 0) {
                            if (!is_numeric($keys)) {
                                $string .= $keys . '=' . $key . '=' . $value . '&';
                                $temp++;
                            } else {
                                $string .= $key . '=' . $value . '&';
                                $temp++;
                            }

                        } else {
                            $string .= $key . '=' . $value . '&';
                        }
                    }
                } else {
                    $string .= $key . '=' . $value . '&';
                }
            }
        }
        return $string;
    }
