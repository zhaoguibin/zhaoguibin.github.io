数组key按照alphabet序排序#
  
    /**
     * @FuncName:arraySoft
     * @Description 数组key按照alphabet序排序
     * @User ZHaoGuiBin
     * @Date 2020-08-13 15:45
     * @param $data
     * @return mixed
     */
    private static function arraySoft($data)
    {
        ksort($data);
        foreach ($data as $key => $value) {
            if (is_array($value)) {
                $data[$key] = self::arraySoft($value);
            }
        }
        return $data;
    }
