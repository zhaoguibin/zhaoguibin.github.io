逢4进5,逢5进10。0,5值不变

January 9, 2018 · 1 min · ZhaoGuibin 
<?php
//逢4进5,逢5进10。0,5值不变
function ceiling($number = 0){

    if($number == 0){
        return 0;
    }

    //这里取出整数
    $number = explode('.',$number);
    $number = $number[0];

    $cc = $number%10;
    $price = $number;
    if($cc > 5){
        $price = $number+(10-$cc);
    }elseif ($cc > 0) {
        $price = $number+(5-$cc);
    }
    return $price;
}
